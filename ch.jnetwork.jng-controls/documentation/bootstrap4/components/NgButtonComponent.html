import { Component} from '@angular/core';
import { NgButtonCommon } from '@jnetwork/jngcontrols-common';

/**
 * Button Component
 * 
 * <example-url>http://localhost/demo/mysample.component.html</example-url>
 * 
 * @example
 * <sac-button (clicked)="action()" text="My Button"></sac-button>
 */
@Component({
  selector: 'sac-button',
  templateUrl: './button.html',
})
export class NgButtonComponent extends NgButtonCommon {

}

<button [id]="name" type="button" class="btn d-flex justify-content-center align-items-center" [attr.disabled]="_isdisabled ? 'disabled' : null" (click)="buttonClick()"
        [class.btn-primary]="role === 'primary'"
        [class.btn-secondary]="role === 'secondary' || role === 'default'"
        [class.btn-success]="role === 'success'"
        [class.btn-danger]="role === 'danger'"
        [class.btn-warning]="role === 'warning'"
        [class.btn-info]="role === 'info'"
        [class.btn-link]="role === 'link'">
  <i *ngIf="icon !== ''" [ngClass]="icon" class="mr-2"></i>{{text}}
</button>
