import { Component } from '@angular/core';
import { NgDateSelectorCommon } from '@jnetwork/jngcontrols-common';

/**
 * DateTime Selektor Komponente
 */
@Component({
  selector: 'ng-dateselector,ngDateSelector',
  templateUrl: './dateselector.html',
})
export class NgDateSelectorComponent extends NgDateSelectorCommon {}

<div class="calendar-selector">
  <div *ngIf="_dateselection">
    <div class="container">
      <div class="row">
        <div class="col text-left"><a (click)="monthBack()"><span class="fa fa-chevron-left"></span></a></div>
        <div class="col text-center">{{_month+1}}/{{_year}}</div>
        <div class="col text-right"><a (click)="monthNext()"><span class="fa fa-chevron-right"></span></a></div>
      </div>
    </div>
    <div></div>
    <div class="container">
      <div class="row">
        <ng-container *ngFor="let week of dates; let i=index">
          <ng-container *ngFor="let item of week">
            <div class="p-2 col text-center" [ngStyle]="{'cursor': item.isenabled ? 'pointer' : null}"
                 [ngClass]="{'day-current': item.iscurrent, 'day-selected': item.isselected, 'day-new': item.isnew}"
                 (click)="selectDate(item)">{{ item.displaytext }}</div>
          </ng-container>
          <div class="w-100"></div>
        </ng-container>
      </div>
    </div>
  </div>
  <div class="d-flex flex-row justify-content-start" *ngIf="_timeselection">
    <div class="p-2">
      <div class="form-group">
        <label>Stunde</label>
        <input type="number" class="form-control form-control-sm" [value]="getHours()" (input)="setHours($event.target.value)" min="0" max="23">
      </div>
    </div>
    <div class="p-2">
      <div class="form-group">
        <label>Minute</label>
        <input type="number" class="form-control form-control-sm" [value]="getMinutes()" (input)="setMinutes($event.target.value)" min="0" max="59">
      </div>
    </div>
  </div>
  <div class="d-flex flex-row">
    <div class="mr-auto p-2">
      <button type="button" class="btn btn-secondary" (click)="setToday()">Heute</button>
    </div>
    <div class="p-2">
      <button type="button" class="btn btn-secondary" (click)="resetSelection()">Reset</button>
    </div>
    <div class="p-2">
      <button type="button" class="btn btn-primary" (click)="applySelection()">Apply</button>
    </div>
  </div>
</div>
