import { Component, forwardRef, Injector } from '@angular/core';
import { NG_VALIDATORS, NG_VALUE_ACCESSOR } from '@angular/forms';
import { NgPagingCommon } from '@jnetwork/jngcontrols-common';


@Component({
  selector: 'sac-paging',
  templateUrl: './paging.html',
  // Value Access Provider registrieren, damit Wert via Model geschrieben und gelesen werden kann
  providers: [
    { provide: NG_VALUE_ACCESSOR, multi: true, useExisting: NgPagingComponent },
    { provide: NG_VALIDATORS, multi: true, useExisting: forwardRef(() => NgPagingComponent) }
  ]
})
export class NgPagingComponent extends NgPagingCommon {

  /**
   * Konstruktor
   */
  constructor(injector: Injector) { super(injector); }

}

<div class="row">
  <div class="pager-nav col-sm-9" id={{name}}>
    <ul class="pagination">
      <li [ngClass]="{'aspNetDisabled disabled': activePageIndex === firstPageIndex}">
        <a (click)="firstPage()"><i class="glyphicon glyphicon-fast-backward"></i></a>
      </li>
      <li [ngClass]="{'aspNetDisabled disabled': activePageIndex === firstPageIndex}">
        <a (click)="previousPage()"><i class="glyphicon glyphicon-step-backward"></i></a>
      </li>
      <li *ngFor="let page of paginators; let i = index" [ngClass]="{'active': page === activePageIndex}">
        <a (click)="changePage(page)">{{page + 1 }}</a>
      </li>
      <li [ngClass]="{'aspNetDisabled disabled': activePageIndex === lastPageIndex}">
        <a (click)="nextPage()"><i class="glyphicon glyphicon-step-forward"></i></a>
      </li>
      <li [ngClass]="{'aspNetDisabled disabled': activePageIndex === lastPageIndex}">
        <a (click)="lastPage()"><i class="glyphicon glyphicon-fast-forward"></i></a>
      </li>
    </ul>
  </div>
  <div class="col-sm-3 text-right" style="margin: 10px 0;">
    Seite {{getCurrentPageNumber()}} von {{getTotalPageNumber()}}
  </div>
</div>
