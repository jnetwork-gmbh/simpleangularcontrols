        *ngIf="nodes && nodes.length">

          class="jstree-node" 
          [ngClass]="{
      'jstree-open': !node[collapseAttr] && node[childrenAttr] && node[childrenAttr].length,
      'jstree-closed': node[collapseAttr]  && node[childrenAttr] && node[childrenAttr].length,
      'jstree-leaf': !node[childrenAttr].length,
      'jstree-last': last
    }" 
    [id]="node[idAttr]" >

      

          <div unselectable="on" role="presentation" class="jstree-wholerow" (mouseover)="node.hover=true" (mouseout)="node.hover=false"

          </div>


          <i class="jstree-icon jstree-ocl" role="presentation" (click)="onCollapseClick(node)" (mouseover)="node.hover=true" (mouseout)="node.hover=false"></i>

            [ngClass]="{'jstree-clicked': node == selectedNode, 'jstreehovered': node.hover}" (mouseover)="node.hover=true" (mouseout)="node.hover=false">

            <ng-container *ngTemplateOutlet="treefileicon; context { item: node }"; content></ng-container>
            
           
            <span>{{node[textAttr]}}</span>
          </a>
          
          
        <ng-container *ngTemplateOutlet="templateTree; context { item: node }"; content></ng-container>
        


          <ngTreeViewChild *ngIf="node[childrenAttr] && node[childrenAttr].length" [fileicontemplate]="fileicontemplate" [template]="templateTree" [collapseAll]="_collapseAll"
